Step 2: Data Models üì¶

Define our Bookmark data structure using dataclasses.

2.1 Create Bookmark Model 10 min
Define the Bookmark dataclass with all fields we need.
models.py 
from dataclasses import dataclass, field
from datetime import datetime
from typing import Optional
from uuid import uuid4


@dataclass
class Bookmark:
    """A saved bookmark."""
    
    title: str
    url: str
    id: str = field(default_factory=lambda: str(uuid4())[:8])
    tags: list[str] = field(default_factory=list)
    description: str = ""
    created_at: str = field(
        default_factory=lambda: datetime.now().isoformat()
    )
    
    def to_dict(self) -> dict:
        """Convert to dictionary for JSON serialization."""
        return {
            "id": self.id,
            "title": self.title,
            "url": self.url,
            "tags": self.tags,
            "description": self.description,
            "created_at": self.created_at
        }
    
    @classmethod
    def from_dict(cls, data: dict) -> "Bookmark":
        """Create Bookmark from dictionary."""
        return cls(
            id=data.get("id", str(uuid4())[:8]),
            title=data["title"],
            url=data["url"],
            tags=data.get("tags", []),
            description=data.get("description", ""),
            created_at=data.get("created_at", datetime.now().isoformat())
        )
üß† Why Dataclasses?
  • Clean syntax: Less boilerplate than regular classes
  • Type hints: Built-in documentation
  • Auto-generated: __init__, __repr__, __eq__ for free
  • Immutable option: Use frozen=True if needed

Search implementation in store.py 
def search(
    self, 
    query: str = "", 
    tag: str = ""
) -> list[Bookmark]:
    """Search bookmarks by query and/or tag."""
    results = list(self.bookmarks.values())
    
    if query:
        query = query.lower()
        results = [
            b for b in results
            if query in b.title.lower() 
            or query in b.url.lower()
            or query in b.description.lower()
        ]
    
    if tag:
        tag = tag.lower()
        results = [
            b for b in results
            if tag in [t.lower() for t in b.tags]
        ]
    
    return results
